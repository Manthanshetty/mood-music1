Frontend (search.js) — replace entirely with:
javascriptconst searchInput = document.getElementById('searchInput');
const resultsContainer = document.getElementById('searchResults');
let debounceTimer;

searchInput.addEventListener('input', function () {
    clearTimeout(debounceTimer);
    const query = this.value.trim();
    if (query.length < 2) {
        resultsContainer.innerHTML = '';
        return;
    }
    debounceTimer = setTimeout(() => {
        fetch(`/search?q=${encodeURIComponent(query)}`, {
            headers: { 'X-Requested-With': 'XMLHttpRequest' }
        })
        .then(res => res.json())
        .then(songs => {
            if (songs.length === 0) {
                resultsContainer.innerHTML = `<div class="text-center text-muted py-4">🎵 No songs found for "<strong>${query}</strong>". Try a different search term.</div>`;
                return;
            }
            resultsContainer.innerHTML = songs.map(s => `
                <div class="song-card d-flex align-items-center justify-content-between p-3 mb-2 rounded glass-card">
                    <div>
                        <strong class="text-white">${s.SongName}</strong>
                        <div class="text-muted small">${s.Artist} · ${s.Duration}</div>
                    </div>
                    <div class="d-flex gap-2 align-items-center">
                        <span class="badge bg-secondary">${s.Language}</span>
                        <span class="badge bg-primary">${s.Genre}</span>
                        <button class="btn btn-sm btn-purple" onclick="playNow('${s.YoutubeID}', '${s.SongName}', '${s.Artist}', '${s.SongID}')">▶ Play</button>
                    </div>
                </div>
            `).join('');
        })
        .catch(err => console.error('Search error:', err));
    }, 300);
});

function playNow(ytId, title, artist, songId) {
    document.getElementById('modalSongTitle').textContent = title;
    document.getElementById('modalArtistName').textContent = artist;
    document.getElementById('modalYTFrame').src = `https://www.youtube.com/embed/${ytId}?autoplay=1&rel=0&modestbranding=1`;
    fetch(`/music/play/${songId}`, { method: 'POST', headers: { 'X-Requested-With': 'XMLHttpRequest' } });
    const modal = new bootstrap.Modal(document.getElementById('playerModal'));
    modal.show();
}
